#+begin_quote
This paper proposes std::ranges::cartesian_product_view for taking the cartesian product of multiple forward ranges.
#+end_quote

** Before
#+begin_src c++
std::vector<int> a,b,c;
for (auto&& ea : a) {
    for (auto&& eb : b) {
        for (auto&& ec : c) {
            use(ea,eb,ec);
        }
    }
}
#+end_src
** After
#+begin_src c++
std::vector<int> a,b,c;
for (auto&& [ea,eb,ec] : std::views::cartesian_product(a,b,c)) {
    use(ea,eb,ec);
}
#+end_src
